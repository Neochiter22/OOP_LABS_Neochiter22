calculator.cpp
    #include "../include/calculator.h"
    using namespace std;

    std::string calculator(int n) {
        string x = "";
        for (int i = 0; i < 32; ++i) {
            if (n % 10 % 2 == 1) {
                x = "1" + x; 
            } else {
                x = "0" + x;
            }
            n >>= 1; 
        }
        return x;
    }
